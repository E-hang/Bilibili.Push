简介

IniCofnig 是基于 正则表达式 应用于 .Net 平台的 Ini配置文件工具类, 在不依赖于 WindowsApi 的同时能快速生成与解析配置文件的 "节"、"键"、"值" 和 "注释", 转换为对象模型, 轻松实现针对 ini 的增删改查.

示例

  • 基础操作示例

    C#

    // 创建 Ini 文件托管实例
IniConfig ini = new IniConfig ("config.ini");

// 1. 加载文件
ini.Load ();

// 2. 增加一个节点
ini.Object.Add (new ISection ("新节点1"));
ini.Object["新节点2"] = new ISection ("新节点2");     // 这里 ISection 的参数必须和前面的 Key 一致

// 3. 删除一个节点
ini.Object.Remove ("新节点1");
ISection removeSection = ini.Object["新节点2"];
ini.Object.Remove (removeSection);

// 4. 为节点新增配置
ini.Object["新节点1"]["StringValue"] = "Value1";       // 若 "新节点1" 不存在会自动创建, StringValue 不存在也会自动创建
ini.Object["新节点1"].Add ("Int32Value", 50);
ini.Object["新节点1"].Add (new KeyValuePair<string, IValue> ("GUIDValue", Guid.NewGuid ()));

// 5. 从节点删除配置
ini.Object["新节点1"].Remove ("StringValue");

// 6. 修改节点的配置
ini.Object["新节点1"]["Int32Value"] = Int32.MaxValue;
ini.Object["新节点1"]["Int32Value"] = new IValue (1000000);

// 7. 获取节点的内容
Guid iniGuid = ini.Object["新节点1"]["GUIDValue"]; // 允许自动转换为 GUID
string iniStr = ini.Object["新节点1"]["GUIDValue"];// 允许自动转换为 String
bool result = ini.Object["新节点1"].TryGetValue ("StringValue", out IValue v);

// 8. 保存文件
ini.Save ();

    C#

    1. 定义模型 
      /// <summary>
/// 描述配置文件中 IP地址 和 端口 的模型类
/// </summary>
//[IniSection (SectionName = "IPConfig")]	// 设置 IniSection 的标记可以设置默认节名称, 未设置则默认使用类名
public class IPAddressModel
{
    public string Host { get; set; }
    public int Port { get; set; }

    [IniKey(KeyName = "PackageSize")]	// 通过设置 IniKey 的标记改变读取或写入时的键
     public long PacketSize { get; set; }
}
    2. 将模型序列化并写入文件 
      IniConfig ini = new IniConfig ("config.ini");
ini.Load ();

// 1. 创建模型数据
IPAddressModel ip = new IPAddressModel
{
    Host = "10.0.0.0",
    Port = 8080,
    PacketSize = 4096
};

// 2. 设置模型到配置文件中, 若存在则修改, 若不存在则创建
// ini.SetObject (ip);  // 设置时会寻找类名或者是类的 IniSection 标记
ini.SetObject ("IPConfig", ip);

// 3. 从文件获取一个节点数据并转化为模型
// ip = ini.GetObject<IPAddressModel> ();	// 读取时会寻找类名或者是类的 IniSection 标记
ip = ini.GetObject<IPAddressModel> ("IPConfig");

ini.Save ();
    VB
    1. 定义模型 
      ''' <summary>
''' 描述配置文件中 IP地址 和 端口 的模型类
''' </summary>
' <IniSection(SectionName:="IPConfig")>   ' 设置 IniSection 的标记可以设置默认节名称, 未设置则默认使用类名
Class IPAddressModel
    Private _host As String
    Public Property Host() As String
        Get
            Return _host
        End Get
        Set(ByVal value As String)
            _host = value
        End Set
    End Property

    Private _port As Integer
    Public Property Port() As Integer
        Get
            Return _port
        End Get
        Set(value As Integer)
            _port = value
        End Set
    End Property

    Private _packetSize As Long
    <IniKey(KeyName:="PackageSize")>    ' 通过设置 IniKey 的标记改变读取或写入时的键
    Public Property PacketSize As Long
        Get
            Return _packetSize
        End Get
        Set(value As Long)
            _packetSize = value
        End Set
    End Property
End Class
